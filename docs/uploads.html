File Uploads

GX wraps multipart file handling behind a clean FileHeader type with helpers for saving, reading, and streaming uploaded files. Content type is detected from magic bytes, not trusted from client headers.

Single file

single_upload.gogo 
func UploadAvatar(c *core.Context) core.Response {
    fh, err := c.FormFile("avatar")
    if err != nil {
        return gx.ErrBadRequest.With("file", "missing avatar field").ToResponse()
    }

    // Security: check content type, not trusting client headers
    if fh.ContentType != "image/jpeg" && fh.ContentType != "image/png" {
        return gx.ErrBadRequest.With("file", "only JPEG and PNG allowed").ToResponse()
    }

    if err := fh.Save("./uploads/" + fh.Filename); err != nil {
        return gx.ErrInternal.Wrap(err).ToResponse()
    }

    return c.JSON(map[string]any{"uploaded": fh.Filename})
}

Multiple files

multi_upload.gogo 
files, err := c.FormFiles("documents")
if err != nil {
    return gx.ErrBadRequest.Wrap(err).ToResponse()
}

for _, fh := range files {
    if err := fh.Save("./uploads/" + fh.Filename); err != nil {
        return gx.ErrInternal.Wrap(err).ToResponse()
    }
}

Reading bytes in memory

When you want to process the file content directly without writing to disk:

read_bytes.gogo 
data, err := fh.Bytes()
if err != nil {
    return gx.ErrInternal.Wrap(err).ToResponse()
}
// process data...

Streaming with Open

For large files, stream via Open() to avoid loading everything into memory:

stream.gogo 
f, err := fh.Open()
if err != nil {
    return gx.ErrInternal.Wrap(err).ToResponse()
}
defer f.Close()

// stream to S3, process with image decoder, etc.
_, err = io.Copy(dst, f)
